PRIMERO CREAR BASE DE DATOS

CREATE DATABASE USUARIOSLEON;


USE USUARIOSLEON;


   

CREATE TABLE INMUEBLES (
    id_inmuebles INT AUTO_INCREMENT PRIMARY KEY,
    DIRECCION VARCHAR(255),
    CODIGO_POSTAL VARCHAR(10),
    BARRIO VARCHAR(255),
    CIUDAD VARCHAR(255)
);



CREATE TABLE EPEC (
    ID_EPEC INT PRIMARY KEY,
    NUMERO_DE_CLIENTE INT,
    NUMERO_DE_CONTRATO INT
);

CREATE TABLE ECOGAS (
    ID_ECOGAS INT PRIMARY KEY,
    NUMERO_DE_CUENTA INT
);

CREATE TABLE AGUAS_CORDOBESAS (
    ID_AGUAS_CORDOBESAS INT PRIMARY KEY,
    UNIDAD VARCHAR(255),
    FACTURACION DECIMAL
);

CREATE TABLE GARANTE (
    ID_GARANTE INT PRIMARY KEY,
    ID_INMUEBLES INT,
    FOREIGN KEY (ID_INMUEBLES) REFERENCES INMUEBLES(ID_INMUEBLES)
);

CREATE TABLE LOCATARIO (
    ID_LOCATARIO INT PRIMARY KEY,
    NOMBRE_Y_APELLIDO VARCHAR(255),
    CONTRASENA VARCHAR(255),
    DNI VARCHAR(15),
    CUIL VARCHAR(15),
    DOMICILIO VARCHAR(255),
    TELEFONO VARCHAR(15),
    CORREO VARCHAR(255),
    OCUPACION VARCHAR(255),
    FK_INMUEBLE INT,
    FOREIGN KEY (FK_INMUEBLE) REFERENCES INMUEBLES(ID_INMUEBLES)
);

CREATE TABLE LOCADOR (
    ID_LOCADOR INT PRIMARY KEY,
    NOMBRE_Y_APELLIDO VARCHAR(255),
 CONTRASENA VARCHAR(255),
    DNI VARCHAR(15),
    CUIL VARCHAR(15),
    DOMICILIO VARCHAR(255),
    TELEFONO VARCHAR(15),
    CORREO VARCHAR(255),
    OCUPACION VARCHAR(255),
    FK_INMUEBLE INT,
    FOREIGN KEY (FK_INMUEBLE) REFERENCES INMUEBLES(ID_INMUEBLES)
);

CREATE TABLE SERVICIOS (
    ID_SERVICIOS INT PRIMARY KEY,
    FK_EPEC INT,
    FK_ECOGAS INT,
    FK_AGUAS_CORDOBESAS INT,
    FOREIGN KEY (FK_EPEC) REFERENCES EPEC(ID_EPEC),
    FOREIGN KEY (FK_ECOGAS) REFERENCES ECOGAS(ID_ECOGAS),
    FOREIGN KEY (FK_AGUAS_CORDOBESAS) REFERENCES AGUAS_CORDOBESAS(ID_AGUAS_CORDOBESAS)
);

CREATE TABLE DEUDAS (
    ID_DEUDA INT PRIMARY KEY,
    ID_INMUEBLE INT,
    CUOTA_ALQUILER INT,
    SERVICIO INT,
    FOREIGN KEY (ID_INMUEBLE) REFERENCES INMUEBLES(ID_INMUEBLES),
    FOREIGN KEY (CUOTA_ALQUILER) REFERENCES CUOTAS_ALQUILER(PK),
    FOREIGN KEY (SERVICIO) REFERENCES SERVICIOS(ID_SERVICIOS)
);
